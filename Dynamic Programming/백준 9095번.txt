n = int(input())
coin = [1, 2, 3]
result = []
for i in range(n):
    a = int(input())
    dp = [1, 1, 2]
    if a < 3:
        result.append(dp[a])
    else:
        for k in range(3, a+1):
            dp.append(dp[k-1]+dp[k-2]+dp[k-3])
        result.append(dp[a])
for i in result:
    print(i, sep='\n')
