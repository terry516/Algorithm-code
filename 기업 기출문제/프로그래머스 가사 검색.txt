#프로그래머스 가사검색 - 카카오문제
#효율성에서 3개가 시간초과 난다... trie를 사용해서 풀어야한다고 한다.
def solution(words, queries):
    answer = []
    for i in queries:
        cnt = 0
        l = len(i)
        c = i.count('?')
        for j in words:
            if l != len(j):
                continue
            else:
                if i[0] != '?':  # ?가 뒤일때
                    if i[0:l-c] == j[0:l-c]:
                        cnt += 1
                else:  # ?가 앞일때
                    if i[c:l] == j[c:l]:
                        cnt += 1
        answer.append(cnt)
    return answer
print(solution(["frodo", "front", "frost", "frozen", "frame", "kakao"],["fro??", "????o", "fr???", "fro???", "pro?"]))
