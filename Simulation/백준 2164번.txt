#deque 이용
from collections import deque
n = int(input())
card = deque()
for i in range(1,n+1):
    card.append(i)
while 1:
    if len(card) == 1:
        break
    else:
        card.popleft()
    if len(card) == 1:
        break
    else:
        card.append(card.popleft())
print(card.pop())
