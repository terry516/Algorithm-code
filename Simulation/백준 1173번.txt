#Baekjoon 1173번 운동
N, m, M, T, R = map(int, input().split())
current_pulse = m
minute = 0
exercise = 0
while exercise < N:
    minute += 1
    if m + T > M:
        minute = -1
        break
    if current_pulse + T <= M:
        exercise += 1
        current_pulse += T
        #print(minute, current_pulse, "¿îµ¿")
    elif current_pulse + T > M:
        if current_pulse - R < m:
            current_pulse = m
        else:
            current_pulse -= R
        #print(minute, current_pulse,"ÈÞ½Ä")
print(minute)
